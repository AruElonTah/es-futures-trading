    PART B — risk.py routes (`packages/api/src/api/routes/risk.py`):

    Create a new FastAPI router with four endpoints. Follow the `backtests.py` pattern
    (APIRouter, Annotated Depends, get_store).

    ```python
    router = APIRouter()

    # GET /positions — UI-05 blotter data source
    @router.get("/positions")
    async def get_positions(request: Request) -> list[dict]:
        """Return in-memory open positions from the FullRiskManager singleton.

        Each position dict: {symbol, strategy_id, side, qty, avg_fill, mark,
        stop, target, entry_ts_utc, engine_state}.
        engine_state field added for frontend bootstrap per UI-SPEC interaction contract.
        """
        rm = getattr(request.app.state, 'risk_manager', None)
        engine_state = getattr(request.app.state.store, None) or 'running'
        # Read engine_state from DuckDB
        engine_state = request.app.state.store.get_engine_state()
        if rm is None:
            return []
        positions = list(getattr(rm, '_positions', {}).values())
        return [dict(p, engine_state=engine_state) for p in positions]

    # POST /kill — SP-05 kill switch
    @router.post("/kill")
    async def kill_switch(request: Request) -> dict:
        """Activate kill switch: halt signal processing. Existing positions held.

        Per D-12: activates even if no positions open.
        Returns: {state: 'killed', positions_held: N}
        """
        store: DuckDBStore = request.app.state.store
        rm = getattr(request.app.state, 'risk_manager', None)
        session_id = getattr(rm, '_session_id', 'unknown')
        store.write_engine_state(session_id=session_id, state='killed')
        # Persist audit event
        from trading_core.storage.runs import new_run_id
        from trading_core.events.models import TOPIC_ENGINE_STATE
        from datetime import datetime, timezone
        store.write_audit_event(
            event_id=new_run_id(),
            ts_utc=datetime.now(timezone.utc),
            topic=TOPIC_ENGINE_STATE,
            entity_id=session_id,
            reason_code='kill_switch',
            payload_json='{"state": "killed"}',
        )
        # Publish WS event so blotter badge updates
        await request.app.state.bus.publish(
            TOPIC_ENGINE_STATE,
            {"type": "engine_state_changed", "payload": {"state": "killed"}},
        )
        n_positions = len(getattr(getattr(rm, '_positions', None) or {}, '__iter__', lambda: []).__self__
                         if hasattr(getattr(rm, '_positions', None), '__len__') else [])
        positions_held = len(rm._positions) if rm and hasattr(rm, '_positions') else 0
        return {"state": "killed", "positions_held": positions_held}

    # POST /flatten — SP-05 flatten command
    @router.post("/flatten")
    async def flatten_all(request: Request) -> dict:
        """Close all open positions at next-bar-open. Per D-12: no-op if no positions.

        Returns: {positions_closed: N}
        """
        store: DuckDBStore = request.app.state.store
        rm = getattr(request.app.state, 'risk_manager', None)
        session_id = getattr(rm, '_session_id', 'unknown')
        positions_held = len(rm._positions) if rm and hasattr(rm, '_positions') else 0
        store.write_engine_state(session_id=session_id, state='flatten_requested')
        store.write_audit_event(
            event_id=new_run_id(),
            ts_utc=datetime.now(timezone.utc),
            topic=TOPIC_ENGINE_STATE,
            entity_id=session_id,
            reason_code='flatten_all',
            payload_json=f'{{"positions_closed": {positions_held}}}',
        )
        await request.app.state.bus.publish(
            TOPIC_ENGINE_STATE,
            {"type": "engine_state_changed", "payload": {"state": "flatten_requested"}},
        )
        # Immediately mark back to running after recording flatten intent
        store.write_engine_state(session_id=session_id, state='running')
        return {"positions_closed": positions_held}

    # POST /pause — toggle pause state
    @router.post("/pause")
    async def toggle_pause(request: Request) -> dict:
        """Toggle engine pause/resume. Does not close positions."""
        store: DuckDBStore = request.app.state.store
        rm = getattr(request.app.state, 'risk_manager', None)
        session_id = getattr(rm, '_session_id', 'unknown')
        current = store.get_engine_state()
        new_state = 'running' if current == 'paused' else 'paused'
        store.write_engine_state(session_id=session_id, state=new_state)
        await request.app.state.bus.publish(
            TOPIC_ENGINE_STATE,
            {"type": "engine_state_changed", "payload": {"state": new_state}},
        )
        return {"state": new_state}
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| POST /kill, /flatten, /pause | No auth in v1 (single operator localhost); all three mutate engine state |
| HWM bootstrap from DuckDB | Read-only query; no injection surface (date is computed from system clock) |
| EodScheduler wall clock | Depends on system TZ and ZoneInfo; DST transitions handled by zoneinfo library |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-04-01 | Spoofing | POST /kill, /flatten without auth | accept | Single-operator localhost; CORS restricted to localhost:3000; no auth surface in v1 |
| T-05-04-02 | Repudiation | kill/flatten not audited | mitigate | Both endpoints write to audit_log via write_audit_event before returning |
| T-05-04-03 | Denial of Service | EodScheduler asyncio.sleep | accept | Sleep target is wall-clock; missed fire (process down) is an accepted risk for paper mode |
| T-05-04-04 | Tampering | CORS method expansion to POST | accept | T-03-05-02 threat model was GET-only; expanding to POST on localhost is acceptable; restrict to specific routes in Phase 7 if needed |
| T-05-04-05 | Information Disclosure | GET /positions leaks position data | accept | Single-operator localhost; CORS restricted; no sensitive data beyond paper fills |
