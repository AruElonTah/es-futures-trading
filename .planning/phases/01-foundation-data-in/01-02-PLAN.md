DataSource Protocol (data/protocols.py):
  - class DataSource(Protocol)
  - name: str
  - async def fetch_bars(self, symbol: str, timeframe: str, start: datetime, end: datetime) -> pd.DataFrame
  - async def subscribe_bars(self, symbol: str, timeframe: str) -> AsyncIterator[Bar]
  - exceptions: DataSourceError, DataSourceUnavailable, RateLimited, GapDetected

Strategy Protocol (strategy/protocols.py):
  - class Strategy(Protocol)
  - name: str; version: str
  - def warmup_bars(self) -> int
  - def on_bar(self, bar: Bar, ctx: "StrategyContext") -> "Signal | None"
  - (StrategyContext + Signal: signature-only stubs in strategy/models.py — full bodies Phase 2)

RiskManager Protocol (risk/protocols.py):
  - class RiskManager(Protocol)
  - async def check(self, signal: "Signal", state: "RiskState") -> "RiskDecision"
  - (RiskState + RiskDecision + RiskConfig: signature-only stubs in risk/models.py — full bodies Phase 5)

Executor Protocol (execution/protocols.py):
  - class Executor(Protocol)
  - async def fill(self, signal: "Signal", decision: "RiskDecision") -> "Fill"
  - (Fill: signature-only stub in execution/models.py — full body Phase 5)

Bar model (data/models.py):
  - class Bar(BaseModel) — frozen=True
  - symbol: str; timeframe: str; ts_utc: AwareDatetime
  - open: Decimal; high: Decimal; low: Decimal; close: Decimal; volume: int (ge=0)
  - rollover_seam: bool = False
  - @field_validator("ts_utc") must_be_utc — rejects offset != 0

Event hierarchy (events/models.py):
  - class Event(BaseModel, frozen=True) — has topic: str; emitted_at: AwareDatetime
  - class BarReceived(Event) — topic = "bars"; bar: Bar
  - class DegradedStateEvent(Event) — topic = "degraded_state"; source: str; reason: str
  - Topic constants (Final[str]): TOPIC_BARS, TOPIC_SIGNALS, TOPIC_RISK_DECISIONS, TOPIC_FILLS, TOPIC_POSITIONS, TOPIC_EQUITY, TOPIC_DEGRADED_STATE
